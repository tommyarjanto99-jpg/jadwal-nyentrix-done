<script>
    // PENTING: Ganti teks di bawah ini dengan Web App URL yang kamu salin dari Google Apps Script
    const API_URL = "https://script.google.com/macros/s/AKfycbwgmKlBX-NmW3g9ojtWjzDk7TatuVLqgwWowMubdQzUxVvzLHjuW-kHPiv0LxkgxV6a/exec";

    // Fungsi Kirim Data ke Google Sheets
    document.getElementById('jadwalForm').addEventListener('submit', function(e) {
        e.preventDefault();
        
        const data = {
            armada: document.getElementById('armada').value,
            tanggal: document.getElementById('tanggal').value,
            pengisi: document.getElementById('pengisi').value,
            nilaiSewa: document.getElementById('nilaiSewa').value,
            penjemputan: document.getElementById('penjemputan').value,
            tujuan: document.getElementById('tujuan').value
        };

        alert("Sedang menyimpan data, mohon tunggu...");

        fetch(API_URL, {
            method: 'POST',
            mode: 'no-cors', // Menghindari masalah CORS di browser
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(data)
        }).then(() => {
            alert("Jadwal berhasil disimpan!");
            document.getElementById('jadwalForm').reset();
            muatJadwal(); // Refresh tabel setelah input
        }).catch(err => alert("Gagal menyimpan: " + err));
    });

    // Fungsi Ambil Data & Filter 2 Bulan ke Depan
    function muatJadwal() {
        fetch(API_URL)
        .then(res => res.json())
        .then(data => {
            const tbody = document.querySelector('#tabelJadwal tbody');
            tbody.innerHTML = '';
            document.getElementById('statusMuat').style.display = 'none';

            // Hitung rentang tanggal (Hari ini sampai 2 bulan ke depan)
            const hariIni = new Date();
            hariIni.setHours(0,0,0,0);
            const duaBulanLagi = new Date();
            duaBulanLagi.setMonth(duaBulanLagi.getMonth() + 2);

            // Urutkan data berdasarkan tanggal terdekat
            data.sort((a, b) => new Date(a.tanggal) - new Date(b.tanggal));

            let dataDitemukan = false;

            data.forEach(item => {
                const tglJadwal = new Date(item.tanggal);
                
                // FILTER: Hanya tampilkan jika masuk dalam rentang 2 bulan ke depan
                if (tglJadwal >= hariIni && tglJadwal <= duaBulanLagi) {
                    dataDitemukan = true;
                    const formatUang = new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', maximumFractionDigits: 0 }).format(item.nilaiSewa);
                    const formatTgl = tglJadwal.toLocaleDateString('id-ID', { day: 'numeric', month: 'short', year: 'numeric' });
                    
                    const row = `<tr>
                        <td><strong>${item.armada}</strong></td>
                        <td>${formatTgl}</td>
                        <td>${item.pengisi}</td>
                        <td>${formatUang}</td>
                        <td>${item.penjemputan}</td>
                        <td>${item.tujuan}</td>
                    </tr>`;
                    tbody.innerHTML += row;
                }
            });

            if (!dataDitemukan) {
                tbody.innerHTML = '<tr><td colspan="6" style="text-align:center;">Belum ada jadwal dalam 2 bulan ke depan.</td></tr>';
            }
        })
        .catch(err => {
            document.getElementById('statusMuat').innerText = "Gagal memuat data.";
            console.error(err);
        });
    }

    // Jalankan fungsi muat data saat web pertama kali dibuka
    muatJadwal();
</script>
